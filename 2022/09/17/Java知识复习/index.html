class User {
    String name;
    int age;

    // 重写其equals()方法
    public boolean equals(Object obj) {
        if (obj == this) {
            return true;
        }

        if (obj instanceof User) {
            User u = (User) obj;
            return this.age == u.age && this.name.equals(u.name);
        }

        return false;
    }
